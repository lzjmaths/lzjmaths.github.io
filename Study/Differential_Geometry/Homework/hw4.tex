\begin{problem}
    Noticed that $\varphi:\mathbb{R}\times \mathbb{R}^2 \rightarrow\mathbb{R}$ is a global flow of vector field if and only if 
    \[\dps\frac{\mathrm{d}\varphi}{\mathrm{d}t}(s,p)=X_{\varphi(s,p)},\, \varphi(0,p)=p\]
    So it suffices to solve an differential equation, which is unqiue if we gives the initial contidion by the previous result.

    Now I give the answer and omit the check.

    \begin{enumerate}
      \item[(1)]   $ \varphi(t,p_1,p_2)=(\dps\frac{1}{2}t^2+p_2t+p_1,t+p_2) $ 
      \item[(2)]  $ \varphi(t,p_1,p_2)=(p_1e^t,p_2e^{2t}) $
      \item[(3)]   $ \varphi(t,p_1,p_2)=(p_1e^t,p_2e^{-t}) $
      \item[(4)]   $ \dps \varphi(t,p_1,p_2)=(\frac{p_1+p_2}{2}e^t+\frac{p_1+p_2}{2}e^{-t},\frac{p_1+p_2}{2}e^t-\frac{p_1+p_2}{2}e^{-t}) $ 
    \end{enumerate}
\end{problem}

\begin{problem}
  For  $ (U,\varphi) $ chart of   $ p $, since  $ \dps\lim_{t\to\infty}\gamma(t)=p $, if we restrict  $ M $ to  $ U $, then we obtain an integral curve on  $ U $,  $ \hat{\gamma}:\mathbb{R}\cong(M,+\infty)\rightarrow U\cong \mathbb{R}^n $.
  
  By proper transformation, WLOG we may assume  $ M=\mathbb{R}^n $.
  
  Then  $ \gamma'(t)=X_{\gamma(t)} $ $ \Rightarrow $
  \begin{align}
    \gamma(n)=\gamma(m)+\int_m^nX_{\gamma(t)}\mathrm{d}t
  \end{align}  
  Since  $ \gamma(n) $ converges to  $ p $,  $ \gamma(t) $ is Cauchy sequence  $\Rightarrow $    $ 0=\dps\lim_{m,n\to\infty}\int_{m}^{n}X_{\gamma(t)}=\lim_{m,n\to\infty}\int_{m}^{n}X_{p} $.
  
  Now if  $ X_p\not=0 $, then  \[0=\lim_{m,n\to\infty}\int_{m}^{n}X_{\gamma(t)}=\lim_{m,n\to\infty}\int_{m}^{n}X_{p}=(n-m)X_p \]

  Therefore,  $ X_p=0 $. 
\end{problem}

\begin{problem}
  Dnote  $ M=\mathbb{R}^n $ 

  Otherwise, suppose  $ \dps\lim\limits_{t\to b}|\gamma(t)|=p<\infty $.

  Take  $ (t_i)\rightarrow b $ from left. Then   $ (\gamma(t_i))\rightarrow p $.

  Then  $ \exists $  $ U $ Nbh of  $ p $, local flow $ \varphi:(-\epsilon,\epsilon)\times U\rightarrow M $. Take  $ n $ large enough \st  $ b-t_n<\epsilon $,  $ \gamma(t_n)\in U $. Then  $ \gamma(-+t_n):(a-t_n,b-t_n)\rightarrow M $,  $ \varphi(-,\gamma(t_n)):(-\epsilon,\epsilon)\rightarrow M $ are both integral curves for  $ X  $ starting at  $ \gamma(t_n) $. By uniqueness, they coincide.

  Let  $ \hat{\gamma}:(a,t_n+\epsilon)\rightarrow M $ be defined by  $ \dps\hat{\gamma}(t)=
      \begin{cases}
          \gamma(t),t\in (a,b)\\
          \varphi(t-t_n,\gamma(t_n)), t\in [b,t_n+\epsilon)
      \end{cases} $  

  Then  $ \hat{\gamma} $ is an integral curve with larger domain, then  $ \gamma $ contradiction with the maxity of  $ \gamma $.
\end{problem}

\begin{problem}
  The global flow $ \varphi:\mathbb{R}\times\mathbb
  R^2\rightarrow \mathbb{R}^2 $  generated by   $ X $ only need satisfy:
  \[\dps\frac{\mathrm{d}\varphi}{\mathrm{d}t}(t,x_1,x_2)=X_{\varphi(t,x_1,x_2)},\, \varphi(0,x_1,x_2)=(x_1,x_2)\]
  Denote  $ \varphi=(\varphi_1,\varphi_2) $.
  
  Then equivalently,
  \(\dps\varphi_1'=f\circ \varphi,\varphi_2'=g\circ\varphi,\varphi(0,-)=\id\)
  Noticed that  $ f,g $ is Lipschitz continuous by the condition, hence, by the Picard-Lindelöf theorem,  $ \varphi $ exists. 
\end{problem}
